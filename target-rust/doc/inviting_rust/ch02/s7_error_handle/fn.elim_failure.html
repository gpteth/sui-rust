  1. 断言用于防御

fn extend_vec(v: &mut Vec<i32>, i: i32) {
    assert!(v.len() == 5);
    v.push(i);
}
fn main() {
    let mut vec = vec![1, 2, 3];
    extend_vec(&mut vec, 4);
    extend_vec(&mut vec, 5);
    extend_vec(&mut vec, 6); // panic!
}
